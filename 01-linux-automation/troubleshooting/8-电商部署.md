# 获取所有项目
harbor_url="http://192.168.1.61"
user="admin"
pass="Harbor12345"
page=1
all_projects=()

while true; do
    projects_json=$(curl -s -u "$user:$pass" "$harbor_url/api/v2.0/projects?page=$page&page_size=100")
    count=$(echo "$projects_json" | jq length)
    if [ "$count" -eq 0 ]; then
        break
    fi
    # 提取项目名称并存入数组
    names=$(echo "$projects_json" | jq -r '.[].name')
    all_projects+=($names)
    ((page++))
done

echo "项目总数: ${#all_projects[@]}"
printf '%s\n' "${all_projects[@]}"

# 查询每个项目下的镜像数量
for project in "${all_projects[@]}"; do
    # 获取该项目下的所有仓库（镜像）
    repo_json=$(curl -s -u "$user:$pass" "$harbor_url/api/v2.0/projects/$project/repositories?page_size=100")
    repo_count=$(echo "$repo_json" | jq length)
    echo "$project: $repo_count 个镜像"
done

#!/bin/bash
harbor_url="http://192.168.1.61"
user="admin"
pass="Harbor12345"

# 获取 gcr.io 项目下所有仓库
repos=$(curl -s -u "$user:$pass" "$harbor_url/api/v2.0/projects/gcr.io/repositories?page_size=100" | jq -r '.[].name')

for repo in $repos; do
    project=$(echo "$repo" | cut -d'/' -f1)
    repo_name=$(echo "$repo" | cut -d'/' -f2-)

done

# 问题 1：cartservice 因权限不足崩溃
解决方案
移除容器和 Pod 级别的 securityContext，并删除 livenessProbe 和 readinessProbe（避免健康检查失败）。
# 移除 securityContext 和探针，并提高内存
kubectl patch deployment cartservice -n default --type='json' -p='[
  {"op": "remove", "path": "/spec/template/spec/containers/0/securityContext"},
  {"op": "remove", "path": "/spec/template/spec/securityContext"},
  {"op": "remove", "path": "/spec/template/spec/containers/0/livenessProbe"},
  {"op": "remove", "path": "/spec/template/spec/containers/0/readinessProbe"},
  {"op": "replace", "path": "/spec/template/spec/containers/0/resources/requests/memory", "value": "2Gi"},
  {"op": "replace", "path": "/spec/template/spec/containers/0/resources/limits/memory", "value": "4Gi"}
]'
