### 1.配置文件
```shell
cat > ~/kube-prometheus-stack/values-override.yaml <<'EOF'
global:
  imageRegistry: "192.168.1.61"

prometheus:
  prometheusSpec:
    image:
      repository: quay.io/prometheus/prometheus
      tag: v2.42.0
    storageSpec:
      volumeClaimTemplate:
        spec:
          storageClassName: local-path
          accessModes: ["ReadWriteOnce"]
          resources:
            requests:
              storage: 50Gi

alertmanager:
  alertmanagerSpec:
    image:
      repository: quay.io/prometheus/alertmanager
      tag: v0.25.0

prometheus-operator:
  image:
    repository: quay.io/prometheus-operator/prometheus-operator
    tag: v0.65.1

prometheus-node-exporter:
  image:
    repository: quay.io/prometheus/node-exporter
    tag: v1.5.0

grafana:
  adminPassword: admin
  persistence:
    enabled: true
    storageClassName: local-path
    size: 10Gi
  image:
    repository: docker.io/grafana/grafana
    tag: 9.5.2
  sidecar:
    image:
      repository: quay.io/kiwigrid/k8s-sidecar
      tag: 1.22.0

kube-state-metrics:
  image:
    repository: registry.k8s.io/kube-state-metrics/kube-state-metrics
    tag: v2.8.2

prometheusOperator:
  admissionWebhooks:
    patch:
      image:
        repository: registry.k8s.io/ingress-nginx/kube-webhook-certgen
        tag: v20221220-controller-v1.5.1-58-g787ea74b6
  # 覆盖 busybox 镜像（如果 admission webhook 使用）
  image:
    busybox:
      repository: library/busybox
      tag: latest
EOF
```

harbor_url="http://192.168.1.61"
user="admin"
pass="Harbor12345"
page=1
all_projects=()

while true; do
    projects_json=$(curl -s -u "$user:$pass" "$harbor_url/api/v2.0/projects?page=$page&page_size=100")
    count=$(echo "$projects_json" | jq length)
    if [ "$count" -eq 0 ]; then
        break
    fi
    # 提取项目名称并存入数组
    names=$(echo "$projects_json" | jq -r '.[].name')
    all_projects+=($names)
    ((page++))
done

echo "项目总数: ${#all_projects[@]}"
printf '%s\n' "${all_projects[@]}"

# 查询每个项目下的镜像数量
for project in "${all_projects[@]}"; do
    # 获取该项目下的所有仓库（镜像）
    repo_json=$(curl -s -u "$user:$pass" "$harbor_url/api/v2.0/projects/$project/repositories?page_size=100")
    repo_count=$(echo "$repo_json" | jq length)
    echo "$project: $repo_count 个镜像"
done

# 在 harbor 节点执行以下脚本（包含动态获取仓库列表和 tags）：
#!/bin/bash
harbor_url="http://192.168.1.61"
user="admin"
pass="Harbor12345"

# 获取 monitoring 项目下所有仓库
repos=$(curl -s -u "$user:$pass" "$harbor_url/api/v2.0/projects/monitoring/repositories?page_size=100" | jq -r '.[].name')

for repo in $repos; do
    project=$(echo "$repo" | cut -d'/' -f1)
    repo_name=$(echo "$repo" | cut -d'/' -f2-)
    echo "===== $repo ====="
    # 使用 Docker Registry API v2 获取 tags
    curl -s -u "$user:$pass" "$harbor_url/v2/${project}/${repo_name}/tags/list" | jq -r '.tags[] // empty'
done

# 检查当前失败 Pod 的镜像地址
kubectl describe pod -n monitoring | grep -A5 "Failed to pull image" | grep "image:"
# 或者直接查看每个失败 Pod 的镜像：
for pod in $(kubectl get pods -n monitoring -o name | grep -E "blackbox|grafana|kube-state-metrics|prometheus-adapter"); do
    echo "===== $pod ====="
    kubectl get $pod -n monitoring -o jsonpath='{.spec.containers[*].image}' && echo
done
