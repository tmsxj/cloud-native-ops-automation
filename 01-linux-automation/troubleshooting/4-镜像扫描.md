```shell
# 拉取 prepare 工具镜像
docker pull goharbor/prepare:v2.9.0
docker tag goharbor/prepare:v2.9.0 192.168.1.61/docker.io/goharbor/prepare:v2.9.0
docker push 192.168.1.61/docker.io/goharbor/prepare:v2.9.0

# 拉取 trivy-adapter 镜像
docker pull goharbor/trivy-adapter-photon:v2.9.0
docker tag goharbor/trivy-adapter-photon:v2.9.0 192.168.1.61/docker.io/goharbor/trivy-adapter-photon:v2.9.0
docker push 192.168.1.61/docker.io/goharbor/trivy-adapter-photon:v2.9.0

# 登录 Harbor 私有仓库
docker login 192.168.1.61 -u admin -p Harbor12345

# 拉取 prepare 镜像
docker pull 192.168.1.61/docker.io/goharbor/prepare:v2.9.0
docker tag 192.168.1.61/docker.io/goharbor/prepare:v2.9.0 goharbor/prepare:v2.9.0

# 拉取 trivy-adapter 镜像
docker pull 192.168.1.61/docker.io/goharbor/trivy-adapter-photon:v2.9.0
docker tag 192.168.1.61/docker.io/goharbor/trivy-adapter-photon:v2.9.0 goharbor/trivy-adapter-photon:v2.9.0

cd /data/harbor
# 编辑 harbor.yml，确保 trivy 部分有 offline_scan: true
vi harbor.yml
trivy:
  skip_update: true      # 跳过数据库更新
  offline_scan: true     # 启用离线扫描

./prepare --with-trivy

编辑 /data/harbor/docker-compose.yml，找到 trivy-adapter 服务，在 environment 部分添加：
environment:
  - TRIVY_SKIP_DB_UPDATE=true
  - TRIVY_OFFLINE_SCAN=true
  - SCANNER_TRIVY_SKIP_DB_UPDATE=True
  - SCANNER_TRIVY_OFFLINE_SCAN=True

docker compose down
docker compose up -d

docker exec trivy-adapter env | grep -E "SKIP_DB|OFFLINE"

# 安装 oras 工具（用于拉取 OCI 制品）
curl -LO https://github.com/oras-project/oras/releases/download/v1.0.0/oras_1.0.0_linux_amd64.tar.gz
tar -xzf oras_1.0.0_linux_amd64.tar.gz -C /usr/local/bin/ oras

# 拉取 Trivy 数据库
oras pull ghcr.io/aquasecurity/trivy-db:2
# 得到 db.tar.gz

scp db.tar.gz root@192.168.1.61:/root/

# 创建容器内数据库目录
docker exec trivy-adapter mkdir -p /home/scanner/.cache/trivy/db

# 复制压缩包到容器
docker cp /root/db.tar.gz trivy-adapter:/tmp/

# 解压到数据库目录
docker exec trivy-adapter tar -xzf /tmp/db.tar.gz -C /home/scanner/.cache/trivy/db/

# 删除临时文件
docker exec trivy-adapter rm /tmp/db.tar.gz

# 修正权限（重要）
docker exec trivy-adapter chown -R 10000:10000 /home/scanner/.cache/trivy/db

# 重启 trivy-adapter 容器
docker restart trivy-adapter

docker exec trivy-adapter trivy image --skip-db-update --offline-scan busybox:latest

问题事项：
docker compose 命令报错	使用 docker compose（无横杠）或升级 docker-compose 版本
环境变量 SCANNER_TRIVY_OFFLINE_SCAN 不生效	检查 docker-compose.yml 中是否拼写正确，重启服务
数据库导入后仍尝试联网更新	确认 TRIVY_SKIP_DB_UPDATE=true 已设置
容器内数据库权限错误	执行 chown -R 10000:10000 修复
扫描时出现 context deadline exceeded	网络超时，离线环境未正确配置，检查 offline_scan 参数

```
