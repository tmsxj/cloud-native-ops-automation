```shell
# 获取镜像列表
cat > ~/scripts/get-images-from-chart.sh <<'EOF'
#!/bin/bash
# ============================================================================
# 从 Helm Chart 中提取所有镜像列表
# 用法：在 Chart 根目录下执行此脚本
# 生成文件：
#   images.txt      - 原始镜像（带引号）
#   images-clean.txt - 去除引号后的纯净镜像列表
# ============================================================================

set -euo pipefail

# 检查当前目录是否包含 Chart.yaml（判断是否为 Chart 根目录）
if [ ! -f Chart.yaml ]; then
    echo "错误：当前目录不是 Helm Chart 根目录（未找到 Chart.yaml）"
    exit 1
fi

echo "正在渲染 Helm Chart 模板..."
# 渲染所有资源并捕获输出
helm template . > rendered.yaml 2>&1

echo "提取镜像列表..."
# 从渲染结果中提取 image: 后面的内容，去除引号，去重排序
grep -E 'image:' rendered.yaml | awk '{print $2}' | sort -u > images.txt

# 生成去除引号的版本
sed 's/"//g' images.txt > images-clean.txt

# 清理临时文件
rm -f rendered.yaml

echo "镜像列表已生成："
echo "  原始文件: images.txt"
echo "  清理后文件: images-clean.txt"
echo "镜像数量: $(wc -l < images-clean.txt)"
EOF

chmod +x ~/scripts/get-images-from-chart.sh
cd ~/kube-prometheus-stack
~/scripts/get-images-from-chart.sh
cat images-clean.txt

cd ~/kube-prometheus-stack
cp values.yaml values.yaml.bak
```

```shell
# 镜像同步脚本
cat > ~/scripts/sync_to_harbor_original.sh <<'EOF'
#!/bin/bash
# ============================================================================
# 镜像同步脚本（保留原始仓库路径，推送到 Harbor 对应项目）
# 用法：./sync_to_harbor_original.sh <镜像名> [限速 KB/s]
# 示例：./sync_to_harbor_original.sh quay.io/prometheus/prometheus:v2.42.0 3000
# ============================================================================

set -euo pipefail

# ---------- 配置区 ----------
HARBOR_ADDR="192.168.1.61"
HARBOR_USER="admin"
HARBOR_PASS="Harbor12345"
SSH_US_ALIAS="us"
SSH_HB_ALIAS="hb"
TMP_HB="/tmp/sync2harbor"
TMP_US="/tmp"
# ---------------------------

if [[ $# -lt 1 ]]; then
    echo "用法: $0 <镜像名> [限速 KB/s]"
    exit 1
fi

IMAGE="$1"
BW_LIMIT=${2:-0}

echo "========== 镜像同步任务开始 =========="
echo "镜像: $IMAGE"
echo "限速: $BW_LIMIT KB/s"

TIMESTAMP=$(date +%Y%m%d%H%M%S)
PACKAGE_NAME="images-${TIMESTAMP}.tar.gz"
US_PACKAGE_PATH="$TMP_US/$PACKAGE_NAME"
HB_PACKAGE_PATH="$TMP_HB/$PACKAGE_NAME"
US_MD5_FILE="$TMP_US/$PACKAGE_NAME.md5"
HB_MD5_FILE="$TMP_HB/$PACKAGE_NAME.md5"

ssh "$SSH_HB_ALIAS" "mkdir -p $TMP_HB"

echo "========== 阶段1：美国服务器处理 =========="
ssh "$SSH_US_ALIAS" "docker version" > /dev/null || { echo "错误：美国服务器 docker 不可用"; exit 1; }

echo ">> 拉取镜像 $IMAGE ..."
if ! ssh "$SSH_US_ALIAS" "docker pull $IMAGE"; then
    echo "拉取失败，终止。"
    exit 1
fi

echo ">> 打包镜像..."
if ssh "$SSH_US_ALIAS" "docker save $IMAGE | gzip > $US_PACKAGE_PATH"; then
    echo "打包成功"
else
    echo "打包失败"
    exit 1
fi

echo ">> 计算 MD5..."
US_MD5=$(ssh "$SSH_US_ALIAS" "md5sum $US_PACKAGE_PATH | cut -d' ' -f1")
echo "   us MD5: $US_MD5"
ssh "$SSH_US_ALIAS" "echo $US_MD5 > $US_MD5_FILE"

echo "========== 阶段2：直传到 Harbor 虚拟机 =========="
RSYNC_OPTS="-avzP"
[[ $BW_LIMIT -gt 0 ]] && RSYNC_OPTS="$RSYNC_OPTS --bwlimit=$BW_LIMIT"

echo ">> 传输打包文件..."
if ssh "$SSH_HB_ALIAS" "rsync $RSYNC_OPTS $SSH_US_ALIAS:$US_PACKAGE_PATH $HB_PACKAGE_PATH"; then
    echo "传输成功"
else
    echo "传输失败"
    exit 1
fi

echo ">> 传输 MD5 文件..."
ssh "$SSH_HB_ALIAS" "rsync -avz $SSH_US_ALIAS:$US_MD5_FILE $HB_MD5_FILE"

echo "========== 阶段3：Harbor 虚拟机处理 =========="
HB_MD5=$(ssh "$SSH_HB_ALIAS" "md5sum $HB_PACKAGE_PATH | cut -d' ' -f1")
US_MD5=$(ssh "$SSH_HB_ALIAS" "cat $HB_MD5_FILE")
if [[ "$HB_MD5" != "$US_MD5" ]]; then
    echo "错误：MD5 校验不一致"
    exit 1
else
    echo "✅ MD5 校验一致"
fi

echo ">> 加载镜像..."
ssh "$SSH_HB_ALIAS" "gunzip -c $HB_PACKAGE_PATH | docker load" || {
    echo "镜像加载失败"
    exit 1
}
echo "✅ 镜像加载成功"

# 提取原始仓库和标签
repo="${IMAGE%:*}"
tag="${IMAGE#*:}"
# 提取项目名（第一个斜杠前部分）和剩余路径
project="${repo%%/*}"
rest="${repo#*/}"
target="$HARBOR_ADDR/$project/$rest:$tag"

echo ">> 登录 Harbor..."
ssh "$SSH_HB_ALIAS" "echo $HARBOR_PASS | docker login -u $HARBOR_USER --password-stdin $HARBOR_ADDR" || {
    echo "登录失败"
    exit 1
}

echo ">> 推送镜像到 $target ..."
ssh "$SSH_HB_ALIAS" "docker tag $IMAGE $target && docker push $target" || {
    echo "推送失败"
    exit 1
}

echo "========== 清理临时文件 =========="
ssh "$SSH_US_ALIAS" "rm -f $US_PACKAGE_PATH $US_MD5_FILE"
ssh "$SSH_HB_ALIAS" "rm -f $HB_PACKAGE_PATH $HB_MD5_FILE"
echo "✅ 清理完成"

echo "========== 任务全部成功！=========="
EOF
chmod +x ~/scripts/sync_to_harbor_original.sh
xargs -I {} ~/scripts/sync_to_harbor_original.sh {} 3000 < images-clean.txt
```
